Restaurant App | Documentation


------------ Home Page ------------




restaurantApp.html
pre:None
post:User welcomed
methods:
get table of restaurants from mysql
display slides with restaurant names, descriptions, and images
link to login.php
link to createRestaurantLogin.php
link to GitHub repository



login.php
pre:Restaurants table exists
post:Admin logged in
methods:
form to log in
input username
input password
submit - run php to check for match in Restaurants table
display message if failed
link to adminFrontPage.php if passed



createRestaurantLogin.php
pre:Restaurants table exists
post:Restaurant and admin log in added to Restaurants table
methods:
form to create new log in
input restaurant name
input username
input password
submit - run php to check add to restaurants table
display message if failed
create Restaurant_Maintenance
create Restaurant_Menu
create Restaurant_OrdersToCook
create Restaurant_Accounting
redirect to adminFrontPage.php if passed
link to login.php



logout.php
pre:Admin logged in
post:Admin logged out
methods:
clear session[username]
redirect to restaurantApp.html



------------ Admin ------------




adminFrontPage.php
pre:Restaurant, username, and password in Restaurants table
post:Admin welcomed
methods:
set session[username] to entered username from login.php
print heading
link to maintenanceFront.php
link to menuAlteration.php
link to kitchenFront.php
link to accountingFront.php
link to customerFrontPage.php



menuAlteration.php
pre:Session set, Restaurant_Menu table exists
post:Restaurant_Menu table modified
methods:
form to add
input info for new item
submit - run MenuAlterationAdd.php
form to update popular
input info to update popular
submit - run MenuAlterationPopular.php
form to delete
input info to delete
submit - run MenuAlterationDelete.php



menuAlterationAdd.php
pre:Form submitted with new info
post:Restaurant_Menu table modified
methods:
check if item already exists
add new row to Menu table on mysql
redirect to menuAlteration.php



menuAlterationDelete.php
pre:Form submitted with info to delete
post:Restaurant_Menu table modified
methods:
delete item from Menu table on mysql
redirect to menuAlteration.php



menuAlterationPopular.php
pre:Form submitted with info to update
post:Restaurant_Menu table modified
methods:
get popular item from Accounting table on mysql
delete popular item from Menu table on mysql
add new popular item from Menu table on mysql
redirect to menuAlteration.php



accountingFront.php
pre:Session set, Restaurant_Accounting table exists
post:Restaurant_Accounting table displayed
methods:
get Accounting table from mysql
print Accounting table
form to sort
submit - run php script to sort
form to sort by time
submit - run php script to sort by time



kitchenFront.php
pre:Session set, Restaurant_OrdersToCook table exists
post:Finished orders removed
methods:
get OrdersToCook table
display OrdersToCook table
form to select items
submit - run kitchen.php
refresh every 30 seconds



kitchen.php
pre:Restaurant_OrdersToCook table exists and form submitted with finished orders
post:Finished orders removed and orders sent to Restaurant_Bill_tableNum table
methods:
get OrdersToCook table
delete each item from OrdersToCook table by id
send each item to Bill table
redirect to kitchenFront.php

------------ Customer ------------




customerFrontPage.php
pre:Session set
post:Customer welcomed
methods:
set session[tableNum] to 0
get Restaurant_Maintenance table from mysql
print restaurant name and description
print welcome message
link to menu.php



menu.php
pre:Session set, Restaurant_Menu table exists
post:Form submitted with table number and ordered items
methods:
print heading
input table number
get Menu table from mysql
print Menu table
input quantities
submit - link to yourOrder.php



yourOrder.php
pre:Form submitted with table number and ordered items
post:Ordered items added to new Restaurant_Cart_tableNum table by table number
methods:
print heading
set session[tableNum] to submitted tableNum
create Cart table (if necessary)
insert ordered menu items into Cart table
redirect to yourOrderFront.php



yourOrder.php
pre:Restaurant_Cart_tableNum table created by table number and orders added
post:Form submitted with special requests
methods:
print heading
get Cart table from mysql
print Cart table
input special requests
submit - link to yourOrderSubmitted.php



yourOrderSubmitted.php
pre:Restaurant_Cart_tableNum created with orders and requests
post:Items sent to Restaurant_OrdersToCook table and Restaurant_Cart_tableNum table cleared
methods:
get Cart table from mysql
get each item from Cart table
insert each item into Restaurant_OrdersToCook table
clear each item from Cart table
print heading
link to enjoyYourMeal.php



enjoyYourMeal.php
pre:Order sent to Restaurant_OrdersToCook table
post:Message displayed
methods:
display message
refresh every 10 seconds
if done:
link to menu.html
link to yourBill.php



yourBill.php
pre:Restaurant_Bill_tableNum table exists
post:Bill displayed and total calculated
methods:
print heading
get Bill table from mysql
calculate subtotal, tax, total
print bill
link to thankYou.php



thankYou.php
pre:Restaurant_Bill_tableNum table exists
post:Order sent to Restaurant_Accounting table
methods:
print heading
get Bill table from mysql
send each item to Accounting table
delete each item from Bill table
